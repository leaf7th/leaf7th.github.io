Zsh

sudo apt install zsh # Install

chsh -s /bin/zsh # Set to defalut shell

sh -c "$(wget https://raw.github.com/robbyrussell/oh-myzsh/master/tools/install.sh -O -)" # On-my-zsh

git clone https://github.com/zsh-users/zsh-syntax-highlighting.git ${ZSH_CUSTOM:-~/.oh-my-zsh/custom}/plugins/zsh-syntax-highlighting # Plugins
mkdir ~/.oh-my-zsh/plugins/incr
wget http://mimosa-pudica.net/src/incr-0.2.zsh -O ~/.oh-myzsh/plugins/incr/incr.plugin.zsh
git clone https://github.com/zsh-users/zsh-autosuggestions ${ZSH_CUSTOM:-~/.oh-myzsh/custom}/plugins/zsh-autosuggestions
sudo apt install autojump

然后 vim .zshrc

修改 plugins=(git zsh-syntax-highlighting)

source ~/.oh-my-zsh/custom/plugins/zsh-autosuggestions/zshautosuggestions.plugin.zsh
source /usr/share/autojump/autojump.sh
source ~/.oh-my-zsh/plugins/incr/incr*.zsh
